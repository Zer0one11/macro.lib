async function searchMacros() { const queryText = document.getElementById('macroName').value.trim(); const ext = document.getElementById('extension').value; const resultsDiv = document.getElementById('results'); const loader = document.getElementById('loader'); if (!queryText) return alert("Введите название!"); resultsDiv.innerHTML = ''; loader.style.display = 'block'; // Формируем поисковый запрос для GitHub const REPO = 'DRANONG/GDMacrosArchive'; const searchQuery = `${queryText} repo:${REPO} path:files/Filtered/ ${ext}`; try { // Вызываем нашу серверную функцию Vercel const response = await fetch(`/api/search?q=${encodeURIComponent(searchQuery)}`); const data = await response.json(); loader.style.display = 'none'; if (data.items && data.items.length > 0) { data.items.forEach(item => { const div = document.createElement('div'); div.className = 'result-item'; const rawUrl = `https://raw.githubusercontent.com/${REPO}/main/${item.path}`; div.innerHTML = ` ${item.name} Папка: ${item.path.split('/').slice(-2, -1)}
🔗 GitHub 📥 Скачать
`; resultsDiv.appendChild(div); }); } else { resultsDiv.innerHTML = '

Ничего не найдено.

'; } } catch (error) { loader.style.display = 'none'; resultsDiv.innerHTML = '

Ошибка сервера.

'; } }
